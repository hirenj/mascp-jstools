 34 /** Build a data retrieval class that uses the given function to extract result data.
 35  *  @static
 36  *  @param  {Function}  dataExtractor   Function to extract data from the resultant data (passed as an argument
 37  *                                      to the function), and then populate the result object. The function is
 38  *                                      bound to a hash to populate data in to. When no data is passed to the
 39  *                                      function, the hash should be populated with default values.
 40  */
 41 MASCP.buildService = function(dataExtractor)
 42 {
 43     var clazz = function(agi,endpointURL)
 44     {
 45         this._endpointURL = endpointURL;
 46         this.agi = agi;
 47         return this;
 48     };
 49 
 50     clazz.Result = function(data)
 51     {
 52         jQuery.extend(this,dataExtractor.apply(this,[data]));
 53     };
 54     
 55     clazz.prototype = jQuery.extend(new MASCP.Service(),{
 56         '__class__'       :       clazz,
 57         '__result_class'  :       clazz.Result,
 58         '_endpointURL'    :       null
 59     });
 60     
 61     clazz.Result.prototype = jQuery.extend(new MASCP.Service.Result(),{
 62        '__class__'        :       clazz.Result
 63     });
 64 
 65     clazz.Result.prototype = jQuery.extend(clazz.Result.prototype,dataExtractor.apply({},[]));
 66     return clazz;
 67 };

158

164             }

167             this.value = "Fetching..";
168             jQuery(readers).each(function() {

170             });
171         });

174

214 MASCP.Service.prototype._dataReceived = function(data,status)
215 {
216     var clazz = this.__result_class;
217 
218     if ( ! this.result ) {
219         result = new clazz(data);
220         this.result = result;
221     } else {
222         var new_result = new clazz(data);
223         jQuery.extend( this.result, new_result );        
224     }
225     this.result.reader = this;
226     this.result.agi = this.agi;
227 };

257 /**
258  *  Asynchronously retrieves data from the remote source. When data is received, a 
259  *  resultReceived.mascp event is triggered upon this service, while an error.mascp
260  *  event is triggered when an error occurs. This method returns a reference to self
261  *  so it can be chained.
262  */
263 MASCP.Service.prototype.retrieve = function()
264 {
265     var self = this;
266     var request_data = this.requestData();
267     if (! request_data ) {
268         return this;
269     }
270     jQuery.ajax(
271         jQuery.extend({
272         async:      this.async,
273         url:        this._endpointURL,
274         error:      function(response,req,settings) {
275                         jQuery(self).trigger("error");
276                         throw "Error occurred retrieving data for service "+self._endpointURL;
277                     },
278         success:    function(data,status) {
279                         self._dataReceived(data,status);
280                         jQuery(self).trigger("resultReceived");
281                         jQuery(MASCP.Service).trigger("resultReceived");
282                     }
283         },request_data)
284     );
285     return this;
286 };

308     if (this.setupSequenceRenderer) {

328  *  Move a node from an externally retrieved document into this current document.
329  *  @static
330  *  @param  {Node}  externalNode    Node from XHR data source that is to be imported into the current document.
331  */
332 MASCP.Service.importNode = function(external_node)
333 {
334     if ( document.importNode ) {
335         return document.importNode(external_node,true);
336     } else {
337         var new_data = jQuery('<div></div>');
338         new_data.innerHTML = external_node.xml;
339         return new_data.firstChild;
