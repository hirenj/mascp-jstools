 78 MASCP.SubaReader.Result.prototype._getLocalisation = function(localisation)
 79 {
 80     var results = {};
 81     var any_data = false;
 82     for (var i = 0; i < this._raw_data['observed'].length; i++) {
 83         var obs = this._raw_data['observed'][i];
 84         if (obs[2] == localisation) {
 85             if (! results[obs[0]]) {
 86                 results[obs[0]] = [];
 87             }
 88             results[obs[0]].push(obs[1]);
 89             any_data = true;
 90         }
 91     }
 92     if ( ! any_data ) {
 93         return null;
 94     }
 95     return results;
 96 };
 97 
 98 MASCP.SubaReader.Result.prototype._parseLocalisation = function(localisation)
 99 {
100     if (localisation == null || localisation.length == 0 )
101     {
102         return null;
103     }
104     var experiments = localisation.split(';');
105     var tissues = {}
106     for (var i = 0; i < experiments.length; i++) {
107         var data = experiments[i].split(':');
108         tissues[data[0]] = tissues[data[0]] || [];
109         tissues[data[0]].push(data[1]);
110     }
111     return tissues;
112 };
113 
114 MASCP.SubaReader.Result.prototype._sortLocalisation = function(loc_data)
115 {
116     var loc_keys = [];
117     for (var i in loc_data) {
118         loc_keys.push(i);
119     }
120     loc_keys = loc_keys.sort(function(a,b) {
121         return loc_data[a].length - loc_data[b].length;
122     });
123     
124     return loc_keys;    
125 };

144 MASCP.SubaReader.Result.prototype.mapController = function(inputElement)
145 {
146     if ( ! this._map ) {
147         return null;
148     }
149     var map = this._map;
150     inputElement = inputElement ? jQuery(inputElement) : jQuery('<div><div class="ms"><input class="ms" type="checkbox"/> MS</div><div class="gfp"><input class="gfp" type="checkbox"/> GFP</div></div>');
151     
152     if ( ! this.getMassSpecLocalisation() )  {
153         jQuery('div.ms', inputElement).css({ 'display': 'none' });
154     } else {
155         var ms_loc = this._sortLocalisation(this.getMassSpecLocalisation());
156         jQuery('input.ms', inputElement).bind('change', function() {
157             if (this.value) {
158                 for (var i in ms_loc) {
159                     map.showKeyword(ms_loc[i], '#ff0000');
160                 }                            
161             } else {
162                 for (var i in ms_loc) {
163                     map.hideKeyword(ms_loc[i], '#ff0000');
164                 }                
165             }
166         }).attr('checked', (ms_loc.length > 0));
167     }
168     if ( ! this.getGfpLocalisation() )  {
169         jQuery('div.gfp', inputElement).css({ 'display': 'none' });
170     } else {
171         var gfp_loc = this._sortLocalisation(this.getGfpLocalisation());
172         jQuery('input.gfp', inputElement).bind('change', function() {
173             if (this.value) {
174                 for (var i in gfp_loc) {
175                     map.showKeyword(gfp_loc[i], '#00ff00');
176                 }                            
177             } else {
178                 for (var i in gfp_loc) {
179                     map.hideKeyword(gfp_loc[i], '#00ff00');
180                 }                
181             }
182         }).attr('checked', (gfp_loc.length > 0));
183     }
184 
185     return inputElement[0];
186 };

188 MASCP.SubaReader.Result.prototype.render = function()
189 {
190     var ms_loc = this._sortLocalisation(this.getMassSpecLocalisation());
191     var gfp_loc = this._sortLocalisation(this.getGfpLocalisation());
192     var container = jQuery('<div><a style="display: block; float: right;" href="http://www.plantenergy.uwa.edu.au/applications/suba/flatfile.php?id='+this.reader.agi+'">SUBA</a></div>');
193     if ( ms_loc.length == 0 && gfp_loc.length == 0 ) {
194         return jQuery('<div>No data</div>');
195     }
196     if (ms_loc.length > 0) {
197         container.append('<div>Proteomics: '+ms_loc.join(', ')+'</div>');
198     }
199     if (gfp_loc.length > 0) {
200         container.append('<div>Flourescent Protein: '+gfp_loc.join(', ')+'</div>');
201     }
202     
203     var do_diagrams = (window.location.search.replace(/^\?/, '').indexOf('drawMap') >= 0);
204     
205     if (typeof GOMap != 'undefined' && do_diagrams) {
206         
207         container.text('');
208         var map = new GOMap.Diagram('cell.svg');
209         var map_container = jQuery('<div style="position: relative; height: 0px; width: 100%; margin-bottom: 2px; overflow: hidden;"></div>');
210         container.append(map_container);
211         map_container.bind('load', function() {
212             for (var i in ms_loc) {
213                 map.showKeyword(ms_loc[i]);
214             }            
215             for (var i in gfp_loc) {
216                 map.showKeyword(gfp_loc[i],'#00ff00');
217             }
218             map_container.css({'height': '100%','overflow':'visible'});
219             map.makeInteractive();
220         });
221         map.appendTo(map_container[0]);
222         container.append('<div style="height: 0px; width: 100%; clear: both; float: none;"></div>');
223         this._map = map;
224     }
225     return container[0];
226 };
