 50 /**
 51  * @class       A diagram that can be marked up with keywords.
 52  * @param       image   Image to be used for the diagram. Either an url to an svg file, an existing object element with a src attribute, or a reference to an SVG element if the SVG has been inlined.
 53  * @author      hjjoshi
 54  * @requires    svgweb
 55  */
 56 GOMap.Diagram = function(image) {
 57 
 58     this._highlighted = {};
 59     this._styles_cache = {};
 60 
 61     var url = null;
 62     if (typeof image == 'string') {
 63         url = image;
 64         image = null;
 65     } else if (image.nodeName && image.nodeName.toLowerCase() == 'object') {
 66         url = image.getAttribute('src') || image.getAttribute('data');
 67     } else if (image.nodeName && image.nodeName.toLowerCase() == 'svg') {
 68         this.element = image;
 69         this._svgLoaded();
 70         var evt = document.createEvent('load');
 71         evt.initEvent('ready',false,true);
 72         image.dispatchEvent(evt);
 73         return;
 74     }
 75 
 76     
 77     this.element = document.createElement('object',true);
 78     
 79     this.element.setAttribute('data',url);
 80     this.element.setAttribute('type','image/svg+xml');
 81     this.element.setAttribute('width','100%');
 82     this.element.setAttribute('height','100%');
 83 
 84     var self = this;
 85     this.element.addEventListener('load',function() {
 86         self.element = (this.contentDocument || this.getAttribute('contentDocument')).rootElement;
 87         
 88         // Make the destroy function an anonymous function, so it can access this new
 89         // element without having to store it in a field
 90         
 91         var svg_object = this;
 92         self.destroy = function() {
 93             svgweb.removeChild(svg_object, svg_object.parentNode);
 94         };
 95 
 96         self._svgLoaded();
 97         if (self._container) {
 98             var evt = document.createEvent('Events');
 99             evt.initEvent('load',false,true);
100             self._container.dispatchEvent(evt);
101         }
102     },false);
103     
104     if (image) {
105         this.appendTo(image.parentNode);
106         image.parentNode.removeChild(image);
107     }
108 };

127 /**
128  * Highlight a given keyword on the diagram
129  * @param {String} keyword  GO keyword to highlight
130  * @param {String} color    CSS color string to use as the highlighting colour. Defaults to #ff0000.
131  */
132 GOMap.Diagram.prototype.showKeyword = function(keyword,color) {
133     var els = this._elementsForKeyword(keyword);
134     
135     if (this._highlighted[keyword] && ! color) {
136         return;
137     }
138     
139     color = color || '#ff0000';
140     
141     this._highlighted[keyword] = true;
142 
143     for (var i = 0; i < els.length; i++ ) {
144         els[i]._highlighted = els[i]._highlighted || {};
145         els[i]._highlighted[color] = true;
146         if (els[i].nodeName == 'path' || els[i].nodeName == 'circle') {
147             this._outlineElement(els[i]);
148         }
149     }
150     var self = this;
151     this._recurse(els, function(el) {
152         self._highlightElement(el);
153     });
154         
155 };
156 
157 /**
158  * Hide a given keyword on the diagram
159  * @param {String} keyword  GO keyword to turn highlighting off for
160  */
161 GOMap.Diagram.prototype.hideKeyword = function(keyword,color) {
162     var els = this._elementsForKeyword(keyword);
163     var self = this;
164 
165     this._highlighted[keyword] = false;
166     
167     this._recurse(els, function(el) {
168         if (color != null && el._highlighted) {
169             el._highlighted[color] = false;
170         } else {
171             el._highlighted = {};
172         }
173         for (var col in el._highlighted) {
174             if (el._highlighted[col] == true) {
175                 self._outlineElement(el);
176                 return;
177             }
178         }
179         self._restoreStyle(el);
180     });
181 };

288 /*
289  * Retrieve all the SVG elements that match the given keyword. The SVG document
290  * should have elements marked up with a keyword attribute.
291  * @param {String}  keyword     Keyword to use to search for elements
292  */
293 GOMap.Diagram.prototype._elementsForKeyword = function(keyword) {
294     var root_svg = this.element;
295     var els = [];
296 
297     if (! GOMap.IE ) {
298         els = this._execute_xpath(root_svg,"//*[@keyword = '"+keyword+"']",root_svg.ownerDocument);
299     } else {
300         var some_els = root_svg.ownerDocument.getElementsByTagNameNS(svgns,'*');
301         for (var i = 0; i < some_els.length; i++) {
302             var el_key = some_els[i].getAttribute('keyword');
303             if (el_key == keyword) {                
304                 els.push(some_els[i]);
305             }
306         }
307     }
308     return els;
309 };

311 /* 
312  * Execute an xpath query upon a document, pulling the results into an array of elements
313  * @param {Element} element Start element
314  * @param {String} xpath Xpath query to execute
315  * @param {Document} document Parent document
316  */
317 GOMap.Diagram.prototype._execute_xpath = function(element, xpath, doc) {
318     var results = [];
319     if (doc.evaluate) {
320         xpath_result = doc.evaluate(xpath,element,null,XPathResult.UNORDERED_NODE_SNAPSHOT_TYPE,null);
321         var i = 0;
322         while ( (a_result = xpath_result.snapshotItem(i)) != null ) {
323             results.push(a_result);
324             i++;
325         }
326     } else {
327         xpath_result = element.selectNodes(xpath);
328         for (var i = 0; i < xpath_result.length; i++ ){
329             results[i] = xpath_result.item(i);
330         }
331     }
332     return results;
333 };
334 /*
335  * Perform a breadth-first traversal of the nodelist.
336  * @param {Array} nodelist Starting list of nodes to perform traversal over
337  * @param {Function} function Callback for this traversal. Callback function takes a single argument, which is the currently inspected node.
338  */
339 GOMap.Diagram.prototype._recurse = function(nodelist,callback) {
340     for (var i = 0; i < nodelist.length; i++) {
341         callback.call(this,nodelist[i]);        
342         if (nodelist[i].childNodes.length > 0) {
343             this._recurse(nodelist[i].childNodes,callback);
344         }
345     }
346 };
347 /*
348  * Cache the old style for this element. We need to cache the style for the element so that we can restore the
349  * element style when it is active.
350  * @param {Element} el Element to store the style for
351  */
352 GOMap.Diagram.prototype._cacheStyle = function(el) {    
353     if ( ! el.id ) {
354         var an_id = 'svg'+(new Date).getTime().toString();
355         el.setAttribute('id',an_id);
356     }
357 
358     if (this._styles_cache[el.id] != null || ! el.style || ! el.id ) {
359         return;
360     }
361     this._styles_cache[el.id] = {
362         'stroke'         : el.style.getPropertyValue('stroke'),
363         'stroke-width'   : el.style.getPropertyValue('stroke-width') || '0px',
364         'opacity'        : el.style.getPropertyValue('opacity'),
365         'fill-opacity'   : el.style.fillOpacity || el.style.getPropertyValue('fill-opacity'),
366         'stroke-opacity' : el.style.strokeOpacity || el.style.getPropertyValue('stroke-opacity')
367     };
368 };
369 
370 /*
371  * Restore the style for an element from the cache
372  * @param {Element} element Element to restore the style for. This element must have cache data.
373  */
374 
375 GOMap.Diagram.prototype._restoreStyle = function(element) {
376     if ( ! element.style ) {
377         return;
378     }
379     if (this._styles_cache[element.id]) {
380         var cacher = this._styles_cache[element.id];
381         var properties = {'stroke-width':null,'opacity':null,'stroke':null,'fill-opacity':null,'stroke-opacity':null};
382         for (var prop in properties) {
383             // We don't set null properties in IE because they cause the wrong styles to be displayed
384             if ( GOMap.IE && ! cacher[prop] ) {
385                 continue;
386             }
387             element.style.setProperty(prop,cacher[prop],null);
388         }
389     }
390 };

392 /*
393  * Draw an outline around the given element
394  * @param {Element} element Element to outline
395  */
396 GOMap.Diagram.prototype._outlineElement = function(element) {
397     if ( ! element.style ) {
398         return;
399     }
400     this._cacheStyle(element);
401     
402     var target_color = this._calculateColorForElement(element);
403     
404     element.style.setProperty('stroke',target_color,null);
405     element.style.setProperty('stroke-width',4,null);
406     element.style.setProperty('stroke-opacity',1,null);
407 };

409 /*
410  * Calculate the color fill. Since there may be more than one color highlighted
411  * on this element, we build the pattern if needed, and return a reference to 
412  * that pattern if a pattern is to be used.
413  */
414 
415 GOMap.Diagram.prototype._calculateColorForElement = function(element) {
416     var pattern = "pat";
417     var total_keywords = 0;
418     for (var col in element._highlighted) {
419         if (element._highlighted && element._highlighted[col] == true) {
420             pattern += "_"+col;
421             total_keywords++;
422         }
423     }
424     if (total_keywords == 1) {
425         return pattern.split(/_/)[1];
426     } else {
427         return this._buildPattern(pattern);
428     }
429     
430 };
431 /*
432  * Create a pattern element under the defs element for the svg. If there isn't a defs element
433  * there already, create one and append it to the document.
434  * @param {String} pattern_name Underscore separated pattern name - each component of the pattern should be represented. e.g. ff0000_00ff00_0000ff
435  * @returns The color name that can be used to reference this pattern
436  * @type String
437  */
438 GOMap.Diagram.prototype._buildPattern = function(pattern_name) {
439     var pattern_els = pattern_name.split('_');
440     this._cached_patterns = this._cached_patterns || {};
441     var cleaned_name = pattern_name.replace(/#/g,'');
442     if (this._cached_patterns[cleaned_name]) {
443         return 'url(#'+cleaned_name+')';
444     }
445     
446     var root_svg = this.element;
447     var defs_el = root_svg.ownerDocument.getElementsByTagNameNS(svgns,'defs')[0];
448 
449     if ( ! defs_el ) {
450         defs_el = document.createElementNS(svgns,'defs');
451         root_svg.appendChild(defs_el);
452     }
453 
454     var new_pattern = document.createElementNS(svgns,'pattern');
455     new_pattern.setAttribute('x','0');
456     new_pattern.setAttribute('y','0');
457     new_pattern.setAttribute('width','30');
458     new_pattern.setAttribute('height','30');
459     new_pattern.setAttribute('viewBox', '0 0 100 100');
460     new_pattern.setAttribute('patternUnits','userSpaceOnUse');
461     new_pattern.setAttribute('patternTransform','rotate(45)');
462     new_pattern.setAttribute('id',cleaned_name);
463     var pattern_width = 100.0 / (pattern_els.length - 1);
464     var start_pos = 0;
465     
466     for (var i = 1; i < pattern_els.length; i++ ) {
467         var a_box = document.createElementNS(svgns, 'rect');
468         a_box.setAttribute('x', start_pos);
469         start_pos += pattern_width;
470         a_box.setAttribute('y', 0);
471         a_box.setAttribute('width', pattern_width);
472         a_box.setAttribute('height', 100);
473         a_box.setAttribute('fill', pattern_els[i]);
474         new_pattern.appendChild(a_box);
475     }
476 
477 
478     defs_el.appendChild(new_pattern);
479     this._cached_patterns[cleaned_name] = true;
480     return 'url(#'+cleaned_name+')';
481 };

483 /* Highlight an element by making it opaque
484  * @param {Element} element Element to make opaque
485  */
486 GOMap.Diagram.prototype._highlightElement = function(element) {
487     // Skip this if we don't have a style or has no id and isn't a group
488     if (! element.style || (! element.id && ! element.nodeName == 'g') ) {
489         return;
490     }
491     
492     this._cacheStyle(element);
493 
494     if (element.nodeName == 'path' || element.nodeName == 'circle') {
495         element.setAttribute('opacity','1');
496         element.style.setProperty('opacity',1,null);
497     }
498     element.setAttribute('fill-opacity','1');
499     element.setAttribute('stroke-opacity','1');
500     element.style.setProperty('fill-opacity',1,null);
501     element.style.setProperty('stroke-opacity',1,null);
502 };
503 
504 /* Go through all the elements in the svg document and force the opacity to be translucent. Since
505  * svgweb doesn't support the referencing of extrinsic stylesheets, we need to go through and 
506  * explicitly set the opacity for all the elements. This is really slow on Internet Explorer.
507  * We've got different behaviour for the different svg element types as they all react differently
508  * to having their opacity set.
509  */
510 GOMap.Diagram.prototype._forceOpacity = function() {
511     var root_svg = this.element;
512     var suspend_id = root_svg.suspendRedraw(5000);
513     var els = root_svg.ownerDocument.getElementsByTagNameNS(svgns,'*');
514     for (var i = 0; i < els.length; i++ ) {
515         if (els[i].nodeName == 'svg') {
516             continue;
517         }
518         if (els[i].parentNode && els[i].parentNode.parentNode && els[i].parentNode.parentNode.nodeName == 'defs') {
519             continue;
520         }
521         if (els[i].nodeName == 'defs' || (els[i].parentNode && els[i].parentNode.nodeName == 'defs')) {
522             continue;
523         }
524         if (els[i].nodeName == 'path') {
525             els[i].setAttribute('opacity','0.4');
526             els[i].style.opacity = 0.4;
527         } else {
528             els[i].setAttribute('fill-opacity','0.3');
529             els[i].setAttribute('stroke-opacity','0.2');
530             if (els[i].style) {
531                 els[i].style.fillOpacity = 0.3;
532                 els[i].style.strokeOpacity = 0.2;
533             }
534         }
535     }
536     root_svg.unsuspendRedraw(suspend_id);
537 };

540 /**
541  * @class       State class for adding panning functionality to an element. Each element that is to be panned needs a new instance
542  *              of the Dragger to store state.
543  * @author      hjjoshi
544  * @requires    svgweb
545  */
546 GOMap.Diagram.Dragger = function() {
547   this.oX = 0;
548   this.oY = 0;
549   this.dX = 0;
550   this.dY = 0;
551   this.dragging = false;
552   this.targetElement = null;
553 };
554 
555 /**
556  * Connect this dragger to a particular element. If an SVG element is given, panning occurs within the bounding box of the SVG, and
557  * the image is shifted by using the currentTranslate property. If a regular HTML element is given, the scrollLeft and scrollTop attributes
558  * are used to move the viewport around. 
559  * @param {Element} targetElement Element to enable panning upon.
560  */
561 GOMap.Diagram.Dragger.prototype.applyToElement = function(targetElement) {
562     var self = this;
563     var svgMouseDown = function(evt) {
564       var positions = mousePosition(evt);
565       self.dragging = true;
566       if (self.targetElement) {
567           self.oX = positions[0];
568           self.oY = positions[1];
569           self.dX = self.targetElement.scrollLeft;
570           self.dY = self.targetElement.scrollTop;
571           evt.preventDefault(true);
572           return;
573       }
574 
575       var p = targetElement.createSVGPoint();
576       var positions = mousePosition(evt);
577       p.x = positions[0];
578       p.y = positions[1];
579 
580       var rootCTM = targetElement.getScreenCTM();
581       self.inverseRootCTM = rootCTM.inverse();
582 
583       p = p.matrixTransform(self.inverseRootCTM);
584 
585       var cur_x = targetElement.currentTranslate.x || targetElement.currentTranslate.getX();
586       var cur_y = targetElement.currentTranslate.y || targetElement.currentTranslate.getY();
587 
588       self.oX = p.x - self.dX - cur_x;
589       self.oY = p.y - self.dY - cur_y;
590 
591       evt.preventDefault(true);
592     };
593     
594     var mousePosition = function(evt) {
595         var posx = 0;
596         var posy = 0;
597         if (!evt) var evt = window.event;
598         if (evt.pageX || evt.pageY) 	{
599             posx = evt.pageX;
600             posy = evt.pageY;
601         } else if (evt.clientX || evt.clientY) 	{
602             posx = evt.clientX + document.body.scrollLeft + document.documentElement.scrollLeft;
603             posy = evt.clientY + document.body.scrollTop + document.documentElement.scrollTop;
604         }
605         if (self.targetElement) {
606             posx = evt.screenX;
607             posy = evt.screenY;
608         }
609         return [ posx, posy ];
610     }
611     
612     var mouseMove = function(evt) {
613         this.style.cursor = 'all-scroll';
614         var positions = mousePosition(evt);
615         if (!self.dragging) {
616            return;
617         }
618         this.style.cursor = '-moz-grabbing';
619         targetElement.scrollLeft = self.dX + (self.oX - positions[0]);
620         targetElement.scrollTop = self.dY + (self.oY - positions[1]);
621         evt.preventDefault(true);
622     };
623     
624     var mouseDown = function(evt) {
625         self.dragging = true;
626         var positions = mousePosition(evt);
627         self.oX = positions[0];
628         self.oY = positions[1];
629         self.dX = targetElement.scrollLeft;
630         self.dY = targetElement.scrollTop;
631         evt.preventDefault(true);
632     };
633     
634     var svgMouseMove = function(evt) {
635         this.style.cursor = 'all-scroll';
636         var positions = mousePosition(evt);
637         if (!self.dragging) {
638             return;
639         }
640         this.style.cursor = '-moz-grabbing';        
641         if (self.targetElement) {
642             self.targetElement.scrollLeft = self.dX + (self.oX - positions[0]);
643             self.targetElement.scrollTop = self.dY + (self.oY - positions[1]);
644             evt.preventDefault(true);
645           return;
646         }
647         evt.preventDefault(true);
648 
649 
650         var p = targetElement.createSVGPoint();
651         var positions = mousePosition(evt);
652 
653 
654 
655         p.x = positions[0];
656         p.y = positions[1];
657         
658         p = p.matrixTransform(self.inverseRootCTM);
659         p.x -= self.oX;
660         p.y -= self.oY;
661 
662         self.dX = p.x;
663         self.dY = p.y;
664 
665         if (p.x > 0 || p.y > 0) {
666             return;
667         }
668 
669         if (targetElement.currentTranslate.setXY) {
670             targetElement.currentTranslate.setXY(p.x,p.y);
671         } else {
672             targetElement.currentTranslate.x = p.x;
673             targetElement.currentTranslate.y = p.y;          
674         }
675     };
676 
677     var mouseOut = function(e) {
678         var toTarget = e.relatedTarget || e.toElement;
679         e.preventDefault(true);
680 
681         while (toTarget != null) {
682             if (toTarget == this) {
683                 return;
684             }
685             toTarget = toTarget.parentNode;
686         }
687         self.dragging = false;
688     };
689     
690     var mouseUp = function(evt) { 
691       self.oX = 0;
692       self.oY = 0;
693       self.dX = null;
694       self.dY = null;
695       self.dragging = false;
696       evt.preventDefault(true);
697     };
698     
699     targetElement.setAttribute('cursor','pointer');    
700     
701     if ( ! targetElement.addEventListener) {
702         targetElement.addEventListener = function(name,func,bool) {
703             this.attachEvent(name,func);
704         }
705     }
706 
707     if (targetElement.nodeName == 'svg') {
708         targetElement.addEventListener('mousedown', svgMouseDown, false);
709         targetElement.addEventListener('mousemove', svgMouseMove, false);        
710         targetElement.addEventListener('mouseup', mouseUp, false);
711     } else {
712         targetElement.addEventListener('mousedown', mouseDown, false);
713         targetElement.addEventListener('mousemove', mouseMove, false);        
714         targetElement.addEventListener('mouseup', mouseUp, false);        
715         targetElement.addEventListener('mouseout',mouseOut, false);
716     }
717 
718 };
719 /**
720  * Given an element that implements a zoom attribute, creates a div that contains controls for controlling the zoom attribute. The
721  * zoomElement must have a zoom attribute, and can fire the zoomChange event whenever the zoom value is changed on the object. The
722  * scrollwheel is connected to this element so that when the mouse hovers over the controls, it can control the zoom using only
723  * the scroll wheel.
724  * @param {Object} zoomElement Element to control the zooming for.
725  * @param {Number} min Minimum value for the zoom attribute (default 0)
726  * @param {Number} max Maximum value for the zoom attribute (default 10)
727  * @param {Number} precision Step precision for the zoom control (default 0.5)
728  * @param {Number} value Default value for this control
729  * @returns DIV element containing the controls
730  * @type Element
731  * @see GOMap.Diagram#event:zoomChange
732  */
733 GOMap.Diagram.addZoomControls = function(zoomElement,min,max,precision,value) {
734     min = min || 0;
735     max = max || 10;
736     precision = precision || 0.5;
737     value = value || min; 
738     
739     var controls_container = document.createElement('div');
740     controls_container.style.height = '30%';   
741     var zoomIn = document.createElement('input');
742     zoomIn.setAttribute('type','button');
743     zoomIn.setAttribute('value','+');
744     var zoomOut = document.createElement('input');
745     zoomOut.setAttribute('type','button');
746     zoomOut.setAttribute('value','-');
747     var reset = document.createElement('input');
748     reset.setAttribute('type','button');
749     reset.setAttribute('value','Reset');
750 
751     controls_container.appendChild(reset);    
752 
753     reset.addEventListener('click',function() {
754         zoomElement.zoom = value;
755     },false);
756     
757     var range = document.createElement('input');
758     range.setAttribute('min',min);
759     range.setAttribute('max',max);
760     range.setAttribute('step',precision);
761     range.setAttribute('value',value); 
762     range.setAttribute('type','range');
763     range.setAttribute('style','-webkit-appearance: slider-vertical; height: 100%; width: 1em; position: absolute; top: 0px; margin-top: 2em; left: 50%; margin-left: -0.5em;');
764 
765     if (range.type == 'range') {
766         
767         range.addEventListener('change',function() {
768             zoomElement.zoom = this.value;
769         },false);
770         zoomElement.registerEvent('zoomChange',function() {
771             range.value = zoomElement.zoom;
772         },false);
773         controls_container.appendChild(range);
774         
775     } else {
776         if (! zoomIn.addEventListener) {
777             var addevlis = function(name,func) {
778                 this.attachEvent(name,func);
779             };
780             zoomIn.addEventListener = addevlis;
781             reset.addEventListener = addevlis;
782             zoomOut.addEventListener = addevlis;        
783         }
784         zoomIn.addEventListener('click',function() {
785             zoomElement.zoom += precision;
786         },false);
787         zoomOut.addEventListener('click',function() {
788             zoomElement.zoom -= precision;
789         },false);
790 
791         controls_container.appendChild(zoomOut);
792         controls_container.appendChild(zoomIn);
793     }
794 
795     this._scrollZoomControls(controls_container,zoomElement,precision);
796 
797     return controls_container;
798 };

800 /**
801  * Connect the scroll wheel to the controls to control zoom
802  */
803 GOMap.Diagram._scrollZoomControls = function(controlElement,target,precision) {
804     var hookEvent = function(element, eventName, callback) {
805       if (typeof(element) == 'string') {
806         element = document.getElementById(element);
807       }
808 
809       if (element == null) {
810         return;
811       }
812 
813       if (element.addEventListener) {
814         if (eventName == 'mousewheel') {
815           element.addEventListener('DOMMouseScroll', callback, false);  
816         }
817         element.addEventListener(eventName, callback, false);
818       } else if (element.attachEvent) {
819         element.attachEvent("on" + eventName, callback);
820       }
821     };
822 
823     var mouseWheel = function(e) {
824       e = e ? e : window.event;
825       var wheelData = e.detail ? e.detail * -1 : e.wheelDelta;
826       if (wheelData > 0) {
827         target.zoom = target.zoom += precision;
828       } else {
829         target.zoom = target.zoom -= precision;
830       }
831       if (e.preventDefault) {
832         e.preventDefault();
833       }
834       return false;
835     };
836 
837     var isFF = false;
838 
839     if (navigator.userAgent.indexOf('Gecko') >= 0) {
840       isFF = parseFloat(navigator.userAgent.split('Firefox/')[1]) || undefined;
841     }                         
842 
843     if (isFF && svgweb.getHandlerType() == 'native') {
844       hookEvent(controlElement, 'mousewheel',
845                 mouseWheel);
846     } else {
847       hookEvent(controlElement, 'mousewheel', mouseWheel);
848     }
849 };
