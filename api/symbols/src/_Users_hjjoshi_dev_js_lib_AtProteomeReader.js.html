 42 /*
 43  * The retrieved ID from the AtProteome database
 44  */
 45 MASCP.AtProteomeIdReader.Result.prototype.getId = function()
 46 {
 47     var doc = this._raw_data;
 48     var trs = doc.getElementsByTagName('tr');
 49     for (var i = 0; i < trs.length; i++) {
 50         if (trs[i].getAttribute('onclick') != null) {
 51             if (trs[i].getAttribute('onclick').match(/protein_id=(\d+)/)) {
 52                 return RegExp.$1;
 53             }
 54         }
 55     }
 56     var content = null;
 57     if (this._raw_data.getElementById) {
 58         content = this._raw_data.getElementById('contentArea');
 59     } else {
 60         content = this._raw_data.selectSingleNode("//*[@id = 'contentArea']");
 61     }
 62     return content || -1;
 63 };

110  */
111 MASCP.AtProteomeTissueReader.Result.prototype.getTotalSpectra = function()
112 {
113     if ( ! this._raw_data ) {
114         return null;
115     }
116     var doc = this._raw_data;
117     var headings = doc.getElementsByTagName('h1');
118     var peptide_table = null;
119     for (var i = 0; i < headings.length; i++) {
120         if (jQuery(headings[i]).text() == 'Peptides') {
121             peptide_table = headings[i].parentNode.getElementsByTagName('table')[1];
122         }
123     }
124     if ( ! peptide_table ) {
125         return null;
126     }
127     var trs = peptide_table.getElementsByTagName('tr');
128     
129     var tds = trs[trs.length-1].getElementsByTagName('td');
130     if (tds.length > 0) {
131         return jQuery(tds[1]).text();
132     } else{
133         return null;
134     }
135 };

138 /*
139  * Get the spectral counts for every peptide in this tissue
140  * @returns Hash mapping peptide position (start-end) to spectra count
141  * @type Hash
142  */
143 MASCP.AtProteomeTissueReader.Result.prototype.getPeptideCounts = function()
144 {
145     if ( ! this._raw_data ) {
146         return null;
147     }
148     var doc = this._raw_data;
149     var headings = doc.getElementsByTagName('h1');
150     var peptide_table = null;
151     for (var i = 0; i < headings.length; i++) {
152         if (jQuery(headings[i]).text() == 'Peptides') {
153             peptide_table = headings[i].parentNode.getElementsByTagName('table')[1];
154         }
155     }
156     if ( ! peptide_table ) {
157         return null;
158     }
159     var trs = peptide_table.getElementsByTagName('tr');    
160     var pep_counts = {};
161     for (var i = 1; i < trs.length - 1; i++) {
162         var tds = trs[i].getElementsByTagName('td');
163         var pep_seq = jQuery(tds[0]).text().replace(/\s/g,'');
164         var spectra_count = parseInt(jQuery(tds[1]).text());
165         var start = parseInt(jQuery(tds[2]).text());
166         var end = start + pep_seq.length;
167         pep_counts[start+"-"+end] = spectra_count;
168     }
169     return pep_counts;
170 }

172 /**
173  * @class   Service class that will retrieve AtProteome data for this entry given an AGI.
174  *          To retrieve data from AtProteome, two requests need to be made to the remote
175  *          server. The first request ascertains the internal ID used by AtProteome for
176  *          the given AGI. A second request then retrieves the data from AtProteome.
177  *          Data is received in XML format, and will need to go through a tidying proxy
178  *          if retrieving data directly from AtProteome.
179  * @description Default class constructor
180  * @param   {String} agi            Agi to look up
181  * @param   {String} endpointURL    Endpoint URL for this service
182  * @extends MASCP.Service
183  */
184 MASCP.AtProteomeReader = MASCP.buildService(function(data) {
185                         this._raw_data = data;
186                         return this;
187                     });

190 /**
191  * Retrieve the protein ID for a given AGI
192  */
193 MASCP.AtProteomeReader.lookupAgi = function(agi,reader)
194 {
195     if (! this.agi_cache || ! this.agi_cache[agi]) {
196         new MASCP.AtProteomeIdReader(agi,reader._endpointURL).bind("resultReceived", function (e) {
197             if ( ! MASCP.AtProteomeReader.agi_cache ) {
198                 MASCP.AtProteomeReader.agi_cache = [];
199             }
200             MASCP.AtProteomeReader.agi_cache[this.agi] = this.result.getId();
201             if (this.result.getId()) {
202                 reader.retrieve();
203             }
204         }).retrieve();
205         return null;
206     }
207     return this.agi_cache[agi];
208 };

211 MASCP.AtProteomeReader.prototype.requestData = function()
212 {
213     var agi = (this.agi+"").replace(/\..*$/,'');
214     
215     var protein_id = MASCP.AtProteomeReader.lookupAgi(agi,this);
216     
217     if ( ! protein_id ) {
218         return null;
219     }
220     if ( protein_id < 0 ) {
221         jQuery(this).trigger('resultReceived');        
222     }
223     
224     var self = this;
225     
226     return {
227         type: "POST",
228         dataType: "xml",
229         data: { 'myassembly': '1#9',
230                 'page'      : 'query_protein',
231                 'queryf'    : agi,
232                 'querystring' : 'protein_id='+protein_id,
233                 'service'   : 'atproteome' 
234         },
235         success: function(data,status) {
236             self._dataReceived(data,status);
237             if (self.result) {
238                 jQuery(self.result).bind('spectrareceived',function() {
239                    jQuery(self).trigger('resultReceived'); 
240                 });
241                 self.result.retrieveSpectraByTissue();
242             }
243         }
244     };
245 };

255 /**
256  * The list of tissue names that are used by AtProteome. This list is
257  * indepedent of spectral data
258  *  @returns {[String]} Tissue names
259  */
260 MASCP.AtProteomeReader.Result.prototype.tissues = function()
261 {
262     if ( ! this._raw_data ) {
263         return null;
264     }
265     var doc = this._raw_data;
266     var headings = doc.getElementsByTagName('h1');
267     var tissue_table = null;
268     for (var i = 0; i < headings.length; i++) {
269         if (jQuery(headings[i]).text() == 'Peptides') {
270             tissue_table = headings[i].parentNode.getElementsByTagName('table')[0];
271         }
272     }
273     if ( ! tissue_table ) {
274         return null;
275     }
276     
277     var table_headings = tissue_table.getElementsByTagName('tr')[0].getElementsByTagName('td');
278     var tissue_names = []
279     for (var i = 0; i < table_headings.length; i++) {
280         if (jQuery(table_headings[i]).text() != "\nAll samples\n") {
281             tissue_names.push(jQuery(table_headings[i]).text());
282         }
283     }
284     return tissue_names;
285 }

306 /**
307  *  Retrieve the spectral data for this result. Fires a {@link MASCP.AtProteomeReader.Result#event:spectrareceived} when the spectra
308  *  are fully populated into the spectra field.
309  */
310 MASCP.AtProteomeReader.Result.prototype.retrieveSpectraByTissue = function()
311 {   
312     var tissues = this.tissues();
313     if ( ! tissues ) {
314         return null;
315     }
316     
317     var this_result = this;
318     
319     
320     this.spectra = {};
321     this.peptide_counts_by_tissue = {}
322     var total_tissues = tissues.length;
323     
324     for (var i in tissues) {
325         this.spectra[tissues[i]] = 0;
326         var tissue_reader = new MASCP.AtProteomeTissueReader(this.agi,this.reader._endpointURL);
327         tissue_reader.tissue = tissues[i];
328         var tissue_name = tissues[i];
329 
330         new function() { 
331             var some_tissue = tissue_name;
332         tissue_reader.bind("resultReceived",function() {
333             total_tissues--;
334             this_result.spectra[some_tissue] = this.result.getTotalSpectra();
335             this_result.peptide_counts_by_tissue[some_tissue] = this.result.getPeptideCounts();
336             if (total_tissues == 0)
337             {
338                 jQuery(this_result).trigger('spectrareceived');
339             }
340         }).retrieve();
341         }();
342     }
343     
344     return null;
345 };

426 MASCP.AtProteomeReader.prototype._normalise = function(array)
427 {
428     var max_val = 0;
429     for (var i = 0; i < array.length; i++)
430     {
431         if (array[i] && array[i] > max_val) {
432             max_val = array[i];
433         }
434     }
435     for (var i = 0; i < array.length; i++)
436     {
437         if (array[i] && array[i] > 0) {
438             array[i] = (array[i] * 1.0) / max_val;
439         }
440     }
441     return array;
442 };

444 MASCP.AtProteomeReader.prototype._mergeCounts = function(hash)
445 {
446     var counts = [];
447     for (var position in hash) {
448         var ends = position.split('-');
449         var start = parseInt(ends[0]);
450         var end = parseInt(ends[1]);
451         for (var i = start; i <= end; i++) {
452             if ( ! counts[i] ) {
453                 counts[i] = 0;
454             }
455             counts[i] += hash[position];
456         }
457     }
458     return counts;
459 };
